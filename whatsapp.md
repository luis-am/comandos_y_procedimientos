#### Comandos Whatsapp

# Nuevo chat

	CTRL + N

# Cerrar cht

	CTRL + W

# Nuevo Grupo

	CTRL + SHIFT + N

# Buscar

	CTRL + F

# Perfil

	CTRL + P

# Silenciar chat

	CTRL + SHIFT + M

# Activar o desactivar confirmación de lectura

	CTRL + SHIFT + U

# Panel de emojis

	CTRL + SHIFT + E

# Panel de GIF

	CTRL + SHIFT + G

# Panel de stickers

	CTRL + SHIFT + S

# Chat anterior

	CTRL + SHIFT + [

# Chat siguiente

	CTRL + TAB

# Abrir chat

	CTRL + 1..9

# Editar el último mensaje

	CTRL + [FLECHA ARRIBA]

# Reducir el tamaño del texto

	CTRL + [-]

# Aumentar el tamaño del texto

	CTRL + [+]

# Reestablecer el tamaño del texto

	CTRL + 0
